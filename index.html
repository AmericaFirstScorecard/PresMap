map.on('load', async () => { try { // Fetch the local JSON file in your repo (no CORS issues!) const res = await fetch('data/election2000.json'); const data = await res.json(); let ev_bush = 0, ev_gore = 0, total_gop_votes = 0, total_dem_votes = 0, total_votes = 0; data.features.forEach(f => { const props = f.properties; if (props.abbr === 'PR') return; const gop = +props.votes_gop || 0; const dem = +props.votes_dem || 0; const reporting = +props.reporting || 0; const total = gop + dem; total_gop_votes += gop; total_dem_votes += dem; total_votes += total; if (reporting === 100) { if (gop > dem) ev_bush += +props.electoral_votes; else if (dem > gop) ev_gore += +props.electoral_votes; } }); const total_registered = 105405100; // Update as needed const perc_gop = ((total_gop_votes / total_registered) * 100).toFixed(1); const perc_dem = ((total_dem_votes / total_registered) * 100).toFixed(1); const reporting_pct = ((total_votes / total_registered) * 100).toFixed(1); document.getElementById('bush-votes').textContent = ev_bush; document.getElementById('gore-votes').textContent = ev_gore; document.getElementById('bush-bar').style.width = `${(ev_bush / 538) * 100}%`; document.getElementById('gore-bar').style.width = `${(ev_gore / 538) * 100}%`; document.getElementById('bush-label').textContent = ev_bush; document.getElementById('gore-label').textContent = ev_gore; document.getElementById('bush-pv').textContent = `Bush: ${total_gop_votes.toLocaleString()} (${perc_gop}%)`; document.getElementById('gore-pv').textContent = `Gore: ${total_dem_votes.toLocaleString()} (${perc_dem}%)`; document.getElementById('reporting').textContent = `Reporting: ${reporting_pct}%`; map.addSource('states', { type: 'geojson', data: data }); map.addLayer({ id: 'states-fill', type: 'fill', source: 'states', paint: { 'fill-color': [ 'match', ['get', 'status'], 'GOP Win', '#d12c0f', 'DEM Win', '#0e47e6', 'Lean GOP', '#f4a6a6', 'Lean DEM', '#a6c8f4', 'Uncalled', '#cccccc', '#ffffff' ], 'fill-opacity': 1.0 } }); map.addLayer({ id: 'states-outline', type: 'line', source: 'states', paint: { 'line-color': '#000000', 'line-width': 2 } }); map.on('click', 'states-fill', (e) => { const p = e.features[0].properties; const gop = +p.votes_gop || 0; const dem = +p.votes_dem || 0; const total = gop + dem; const gopPerc = total > 0 ? ((gop / total) * 100).toFixed(1) : 0; const demPerc = total > 0 ? ((dem / total) * 100).toFixed(1) : 0; const winner = p.winner === 'UNKNOWN' ? 'Uncalled' : p.winner; const gopName = p.gop_candidate || 'Bush'; const demName = p.dem_candidate || 'Gore'; const gopCheck = winner === gopName ? '✓ ' : ''; const demCheck = winner === demName ? '✓ ' : ''; new mapboxgl.Popup() .setLngLat(e.lngLat) .setHTML(` ${p.NAME}
Electoral Votes: ${p.electoral_votes}
Reporting: ${p.reporting}%
${gopCheck}${gopName}: ${gop.toLocaleString()} (${gopPerc}%)
${demCheck}${demName}: ${dem.toLocaleString()} (${demPerc}%)
Winner: ${winner} `) .addTo(map); }); map.on('mouseenter', 'states-fill', () => { map.getCanvas().style.cursor = 'pointer'; }); map.on('mouseleave', 'states-fill', () => { map.getCanvas().style.cursor = ''; }); } catch (error) { console.error('Map load error:', error); document.getElementById('map').innerHTML = '

Failed to load map. Please check data source or internet connection.
